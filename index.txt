    <script>
        let topSpeed = 0;
        let currentSpeed = 0;
        let currentGrad = 0;
        let pitchOffset = 0;
        let currentRawPitch = 0;
        let altHistory = new Array(50).fill(0); 
        let isReacting = false;
        
        const themes = ['retro-green', 'monochrome', 'blue', 'orange', 'red'];
        let themeIdx = 0;

        // EXPANDED QUOTES AND JOKES
        const quotes = [
            "SYSTEM ONLINE", "CRUISING MODE", "STAY VIGILANT", "FUEL CHECK?", 
            "ROAD TRIP!!!", "VIBE CHECK: OK", "SCANNING HORIZON", "ARE WE THERE YET?",
            "DON'T MAKE ME TURN THIS CAR AROUND", "CALCULATING THE MEANING OF LIFE...",
            "LOOK AT THAT GRADIENT!", "ALTITUDE IS ATTITUDE", "I'M NOT LOST, I'M EXPLORING",
            "SENSORS TINGLING", "GPS SAYS: GO THAT WAY", "ERROR: ADVENTURE OVERLOAD",
            "DRIVE IT LIKE YOU STOLE IT (LEGALLY)", "I LIKE YOUR DRIVING STYLE",
            "BEWARE OF THE SPEED TRAP", "WEATHER IS JUST A STATE OF MIND",
            "GRAVITY IS STRONGER TODAY", "FAST AND FURIOUS? MOSTLY FAST",
            "KEEP YOUR EYES ON THE ROAD, HUMAN", "I SEE CLOUDS. NICE CLOUDS.",
            "IS IT LUNCH TIME YET?", "1.21 GIGAWATTS?!", "INITIATING MAXIMUM VIBES",
            "SIRI IS MY COUSIN", "WIND RESISTANCE IS FUTILE", "WATCH THE TURNS!",
            "I'M BUILT DIFFERENT", "COFFEE BREAK SOON?", "SMOOTH OPERATOR",
            "DETECTING A HIGH FREQUENCY OF COOLNESS", "MY BATTERY IS MY SOUL",
            "WHY DID THE CHICKEN CROSS THE ROAD? TO GET TO THE DASHBOARD."
        ];

        const randomFaces = ["(•‿•)", "(¬_¬)", "(⊙_☉)", "(-_-)zzz", "(o_O)!", "(^_-)", "p_p", "(◕‿◕)", "(≖_≖)", "(͡° ͜ʖ ͡°)", "ʕ•ᴥ•ʔ", "ಠ_ಠ"];

        function updateClock() {
            const now = new Date();
            document.getElementById('clock').innerText = now.toTimeString().split(' ')[0];
        }
        setInterval(updateClock, 1000);

        function triggerComms() {
            const container = document.getElementById('quote-container');
            container.innerHTML = '';
            const span = document.createElement('span');
            span.className = 'scroll-once';
            span.innerText = quotes[Math.floor(Math.random() * quotes.length)];
            container.appendChild(span);
        }

        function randomExpression() {
            if (isReacting || currentSpeed > 5) return; 
            isReacting = true;
            const el = document.getElementById('avatar');
            el.innerText = randomFaces[Math.floor(Math.random() * randomFaces.length)];
            
            setTimeout(() => {
                isReacting = false;
                updateAvatar(currentSpeed, currentGrad);
            }, 3000); 
        }

        setInterval(() => {
            if(Math.random() > 0.6) randomExpression(); // Increased frequency slightly
        }, 15000);

        async function fetchWeather(lat, lon) {
            try {
                const response = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,weather_code,wind_speed_10m&temperature_unit=fahrenheit&wind_speed_unit=mph`);
                const data = await response.json();
                
                document.getElementById('temp-val').innerText = Math.round(data.current.temperature_2m);
                document.getElementById('wind-val').innerText = Math.round(data.current.wind_speed_10m);
                document.getElementById('weather-icon').innerText = getWeatherEmoji(data.current.weather_code);
            } catch (e) { console.error("Weather fetch failed", e); }
        }

        function getWeatherEmoji(code) {
            if (code === 0) return '☀️ CLEAR';
            if (code <= 3) return '⛅ CLOUD';
            if (code <= 48) return '🌫️ FOG';
            if (code <= 67) return '🌧️ RAIN';
            if (code <= 77) return '❄️ SNOW';
            if (code <= 82) return '🚿 SHOWER';
            if (code <= 99) return '⚡ STORM';
            return '❓ UNK';
        }

        function drawAltGraph() {
            const canvas = document.getElementById('altGraph');
            const ctx = canvas.getContext('2d');
            const color = getComputedStyle(document.documentElement).getPropertyValue('--text-color').trim();
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
            ctx.strokeStyle = color;
            ctx.lineWidth = 3;
            ctx.beginPath();
            const step = canvas.width / (altHistory.length - 1);
            const minAlt = Math.min(...altHistory) - 5;
            const maxAlt = Math.max(...altHistory) + 5;
            const range = maxAlt - minAlt;
            for(let i = 0; i < altHistory.length; i++) {
                const x = i * step;
                const y = canvas.height - ((altHistory[i] - minAlt) / (range || 1) * canvas.height);
                if(i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
            }
            ctx.stroke();
        }

        function getCardinal(angle) {
            const directions = ['N', 'NE', 'E', 'SE', 'S', 'SW', 'W', 'NW'];
            return directions[Math.round(angle / 45) % 8];
        }

        function updateAvatar(speed, grad) {
            if (isReacting) return; 
            const el = document.getElementById('avatar');
            if (Math.abs(grad) > 15) el.innerText = ">_<";
            else if (speed > 70) el.innerText = "X_X"; 
            else if (speed > 55) el.innerText = "O_O";
            else if (speed > 40) el.innerText = "o_o";
            else if (speed > 30) el.innerText = "^-^";
            else if (grad > 7) el.innerText = "~_~";
            else if (grad < -7) el.innerText = "O_O";
            else el.innerText = "~_~";
        }

        function initSensors() {
            navigator.geolocation.watchPosition((pos) => {
                const lat = pos.coords.latitude;
                const lon = pos.coords.longitude;
                currentSpeed = Math.round((pos.coords.speed || 0) * 2.237);
                const alt = Math.round((pos.coords.altitude || 0) * 3.28);
                const heading = pos.coords.heading;
                
                document.getElementById('speed').innerText = currentSpeed;
                document.getElementById('alt').innerText = alt;
                if(heading !== null) document.getElementById('direction').innerText = getCardinal(heading);

                altHistory.push(alt);
                altHistory.shift();
                drawAltGraph();

                if (currentSpeed > topSpeed) {
                    topSpeed = currentSpeed;
                    document.getElementById('top-speed').innerText = topSpeed;
                }
                
                if (!window.lastWeatherFetch || Date.now() - window.lastWeatherFetch > 600000) {
                    fetchWeather(lat, lon);
                    window.lastWeatherFetch = Date.now();
                }

                if(Math.random() > 0.90) triggerComms(); // Increased frequency
                updateAvatar(currentSpeed, currentGrad);

            }, null, { enableHighAccuracy: true });

            window.addEventListener('deviceorientation', (e) => {
                currentRawPitch = e.beta || 0;
                currentGrad = Math.round(Math.tan((currentRawPitch - pitchOffset) * (Math.PI/180)) * 100);
                document.getElementById('grad').innerText = Math.min(Math.max(currentGrad, -99), 99);

                const roll = e.gamma || 0;
                const avatar = document.getElementById('avatar');
                if (!isReacting) {
                    if (roll > 15) avatar.innerText = ">_>"; 
                    else if (roll < -15) avatar.innerText = "<_<"; 
                    else updateAvatar(currentSpeed, currentGrad);
                }
            });
        }

        function startTracking() {
            if (typeof DeviceOrientationEvent !== 'undefined' && typeof DeviceOrientationEvent.requestPermission === 'function') {
                DeviceOrientationEvent.requestPermission().then(s => { if(s==='granted') initSensors(); });
            } else { initSensors(); }
            document.getElementById('startBtn').innerText = "🛰️ ACTIVE";
            triggerComms();
        }

        function cycleTheme() {
            document.body.classList.remove(themes[themeIdx]);
            themeIdx = (themeIdx + 1) % themes.length;
            document.body.classList.add(themes[themeIdx]);
            setTimeout(drawAltGraph, 10);
        }

        function calibrateLevel() { pitchOffset = currentRawPitch; }
        function resetAll() { topSpeed = 0; document.getElementById('top-speed').innerText = 0; altHistory.fill(0); drawAltGraph(); }
        
        window.addEventListener('resize', drawAltGraph);
        drawAltGraph();
